Astro file
---
import { Navbar } from 'astro-bootstrap';
// demonstration data
const siteTitle = 'Astro site'; // pull this in from a config file
const items = [
  // pull this in from a config file
  { title: 'Home', href: '#' },
  { title: 'About', href: '#' },
  {
    title: 'Dropdown',
    children: [
      { title: 'First child', href: '#' },
      { divider: true },
      { title: 'Second child', href: '#' },
    ],
  },
];
---

<nav class="navbar navbar-expand-lg bg-primary" data-bs-theme="dark">
  <div class="container-fluid">
    <a class="navbar-brand" href="#">{siteTitle}</a>
    <Navbar.Toggler controls="navbarSupportedContent" />
    <Navbar.Collapse id="navbarSupportedContent">
      <ul class="navbar-nav me-auto mb-2 mb-lg-0">
        {
          items.map((item) =>
            item.children ? (
              <Navbar.Dropdown {...item} />
            ) : (
              <Navbar.Item {...item} />
            ),
          )
        }
      </ul>
    </Navbar.Collapse>
  </div>
</nav>

API

Navbar.Toggler

import { Navbar } from 'astro-bootstrap'
NameTypeDefaultDescription
controls required string Must be the same as the id property of the &lt;Navbar.Collapse /&gt; component.

Navbar.Collapse

import { Navbar } from 'astro-bootstrap'
NameTypeDefaultDescription
id required string Must be the same as the contols property of the &lt;Navbar.Toggler /&gt; component.

Navbar.Item

import { Navbar } from 'astro-bootstrap'
NameTypeDefaultDescription
href required string Set the href for the menu item.
title required string Set the innerText for the menu item.

Navbar.Dropdown

import { Navbar } from 'astro-bootstrap'
NameTypeDefaultDescription
title required string Set the innerText for the dropdown menu item.
children required array of objects See below for object properties

Navbar.Dropdown Child

import { Navbar } from 'astro-bootstrap'
NameTypeDefaultDescription
href required string Set the href for the dropdown menu child item.
title required string Set the innerText for the dropdown menu child item.
